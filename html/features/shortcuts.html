Configuration

mode: classic · shortcuts 
var config = Object.assign({}, synapEditorLicense, synapEditorConfig, {
    'editor.toolbar': [
        'new', '|',
        'undo', 'redo', '|',
        'bold', 'italic', 'underline', 'strike', '|',
        'copyRunStyle', 'pasteRunStyle', 'removeRunStyle', '|',
        'decreaseIndent', 'increaseIndent', 'alignLeft', 'alignCenter', 'alignRight', '|',
        'bulletList', 'numberedList', '|',
        'link', 'find', 'print', '|',
        'keyboard'
    ]
});

// Create editor instance
var editor = new SynapEditor('synapEditor', config, initialHtml);
