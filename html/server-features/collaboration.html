Configuration

server: collabo-ws · plugin: collaboration 
// Requires plugins/collaboration to be loaded (see <head>).
var config = Object.assign({}, synapEditorLicense, synapEditorConfig, {
    'editor.size.width': '100%',
    'editor.size.height': '100%',
    'editor.toolbar': ['bold', 'italic', 'underline', '|', 'bulletList', 'numberedList', '|', 'collaborator'],
    'collaboration.config': { wsUrl: '/collabo_ws', useHistory: true }
});

var editor = new SynapEditor('synapEditorA', config, '');

// When ready: identify the user and join a shared document
window.SynapEditorInitialized = function (e) {
    var collab = e.editor.plugins.collaboration;
    collab.setUser({ id: 'user-a', name: 'User A' });
    collab.changeDocument('demo-doc-1');
};
