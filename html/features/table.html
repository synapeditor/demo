Configuration

mode: classic · table tools 
// Formulas / number formats / conditional formatting need these externals in <head>,
// loaded BEFORE synapeditor.min.js:
//   externals/formulaParser/formula-parser.min.js
//   externals/SEDocModelParser/SEDocModelParser.min.js
var config = Object.assign({}, synapEditorLicense, synapEditorConfig, {
    'editor.toolbar': [
        'new', '|',
        'table', 'tableStyle', 'tableProperties', 'conditionalFormatting', '|',
        'fill', 'borderColor', 'lineStyle', '|',
        'mergeCell', 'splitCell'
    ],
    'editor.table.handle': true,
    'editor.menu.show': true
});

// Create editor instance
var editor = new SynapEditor('synapEditor', config, initialHtml);
