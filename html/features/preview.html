Configuration

mode: classic · codeBlock + preview 
// Load the code-block highlighter plugin + a theme in <head> for syntax colors:
// .../plugins/codeBlockHighlighter/codeBlockHighlighter.js  +  themes/github-dark.min.css
var config = Object.assign({}, synapEditorLicense, synapEditorConfig, {
    'editor.toolbar': [
        'new', '|',
        'codeBlock', '|',
        'preview'
    ],
    // Highlight code blocks in the Preview window too
    'editor.preview.style.urls': [
        'https://cdn.synapeditor.com/latest/plugins/codeBlockHighlighter/themes/github-dark.min.css'
    ]
});

// Create editor instance
var editor = new SynapEditor('synapEditor', config, initialHtml);
