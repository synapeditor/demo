Server Feature

Export

Build content by importing a document or inserting images, then export it to Word/HWP to verify the result round-trips correctly.

server: converter-api · export-api api: /importDoc · /uploadFile · /exportFile

Round-trip Test

Use Open to import a .docx/.xlsx/.pptx, or Image to upload a picture. Then click Export to download it as .docx and confirm the content (including images) is preserved.

Configuration

import + upload + export · external: SEDocModelParser 
// Requires externals/SEDocModelParser to be loaded (see <head>).
var config = Object.assign({}, synapEditorLicense, synapEditorConfig, {
    'editor.toolbar': ['new', 'open', '|', 'image', '|', 'export'],
    'editor.import.api': '/importDoc',
    'editor.import.extensions': ['docx', 'xlsx', 'pptx', 'html', 'txt'],
    'editor.upload.image.api': '/uploadFile',
    'editor.upload.video.api': '/uploadFile',
    'editor.upload.file.api': '/uploadFile',
    'editor.export.api': '/exportFile',
    'editor.export.extensions': ['docx', 'hwpx', 'hwp'],
    'editor.export.defaultExtension': 'docx',
    'editor.menu.show': false
});

// Create editor instance
var editor = new SynapEditor('synapEditor', config, initialHtml);
